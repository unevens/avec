289 InterleavedBuffer<Scalar>::SetNumChannels(int value)
290 {
291  numChannels = value;
292  if constexpr (VEC8_AVAILABLE) {
293  auto d8 = std::div(numChannels, 8);
294 #if AVEC_MIX_VEC_SIZES
295  buffers8.resize((std::size_t)d8.quot + (d8.rem > 4 ? 1 : 0));
296 
297  buffers4.resize((d8.rem <= 4 && d8.rem > 0) ? 1 : 0);
298  buffers2.resize(0);
299 #else
300  buffers8.resize((std::size_t)d8.quot + (d8.rem > 0 ? 1 : 0));
301  buffers4.resize(0);
302  buffers2.resize(0);
303 #endif
304  }
305  else if constexpr (VEC4_AVAILABLE) {
306  auto d4 = std::div(numChannels, 4);
307  buffers8.resize(0);
308  buffers4.resize((std::size_t)d4.quot + (d4.rem > 0 ? 1 : 0));
309  buffers2.resize(0);
310  }
311  else {
312  auto d2 = std::div(numChannels, 2);
313  buffers2.resize((std::size_t)d2.quot + (d2.rem > 0 ? 1 : 0));
314  buffers8.resize(0);
315  buffers4.resize(0);

336 template<typename Scalar>
337 bool
338 InterleavedBuffer<Scalar>::Deinterleave(Scalar** output,
339  int numOutputChannels,
340  int numOutputSamples) const
341 {
342  if (numOutputChannels > numChannels || numOutputSamples > numSamples) {
343  return false;
344  }
345  int b = 0;
346  for (int c = 0; c < numOutputChannels; ++c) {
347 
348  if constexpr (VEC8_AVAILABLE) {
349  auto d = std::div(b, 8);
350 #if AVEC_MIX_VEC_SIZES
351  assert(d.quot <= buffers8.size());
352  if (d.quot == buffers8.size()) {
353  assert(buffers4.size() > 0);
354  for (int i = 0; i < numOutputSamples; ++i) {
355  output[c][i] = buffers4[0](4 * i + d.rem);
356  }
357  }
358  else {
359  for (int i = 0; i < numOutputSamples; ++i) {
360  output[c][i] = buffers8[d.quot](8 * i + d.rem);
361  }
362  }
363 #else
364  assert(d.quot < buffers8.size());
365  for (int i = 0; i < numOutputSamples; ++i) {
366  output[c][i] = buffers8[d.quot](8 * i + d.rem);
367  }
368 #endif
369  }
370 
371  else if constexpr (VEC4_AVAILABLE) {
372  auto d = std::div(b, 4);
373  assert(d.quot < buffers4.size());
374  for (int i = 0; i < numOutputSamples; ++i) {
375  output[c][i] = buffers4[d.quot](4 * i + d.rem);
376  }
377  }
378  else {
379  auto d = std::div(b, 2);
380  assert(d.quot < buffers2.size());
381  for (int i = 0; i < numOutputSamples; ++i) {
382  output[c][i] = buffers2[d.quot](2 * i + d.rem);
383  }
384  }
385  ++b;
386  }
387  return true;

390 template<typename Scalar>
391 bool
392 InterleavedBuffer<Scalar>::Interleave(Scalar* const* input,
393  int numInputChannels,
394  int numInputSamples)
395 {
396 
397  if (numInputChannels > numChannels) {
398  SetNumChannels(numInputChannels);
399  }
400  SetNumSamples(numInputSamples);
401 
402  if (VEC8_AVAILABLE && buffers8.size() > 0) {
403  if (numInputChannels % 8 != 0) {
404  Fill(0.f);
405  }
406  }
407  else if (VEC4_AVAILABLE && buffers4.size() > 0) {
408  if (numInputChannels % 4 != 0) {
409  Fill(0.f);
410  }
411  }
412  else {
413  if (numInputChannels % 2 != 0) {
414  Fill(0.f);
415  }
416  }
417 
418  int processedChannels = 0;
419  if constexpr (VEC8_AVAILABLE) {
420  auto d8 = std::div(numInputChannels, 8);
421 
422  for (int b = 0; b < std::min(d8.quot + (d8.rem > 0), (int)buffers8.size());
423  ++b) {
424  int r = std::min(8, numInputChannels - processedChannels);
425  for (int i = 0; i < r; ++i) {
426  auto c = b * 8 + i;
427  for (int j = 0; j < numInputSamples; ++j) {
428  buffers8[b](j * 8 + i) = input[c][j];
429  }
430  }
431  processedChannels += r;
432  assert(processedChannels <= numInputChannels);
433  if (processedChannels == numInputChannels) {
434  return true;
435  }
436  }
437  }
438 
439  if constexpr (VEC4_AVAILABLE) {
440  auto d4 = std::div(numInputChannels, 4);
441  for (int b = 0;
442  b < std::min(d4.quot + (d4.rem > 0 ? 1 : 0), (int)buffers4.size());
443  ++b) {
444  int r = std::min(4, numInputChannels - processedChannels);
445  for (int i = 0; i < r; ++i) {
446  auto c = b * 4 + i;
447  for (int j = 0; j < numInputSamples; ++j) {
448  buffers4[b](j * 4 + i) = input[c][j];
449  }
450  }
451  processedChannels += r;
452  assert(processedChannels <= numInputChannels);
453  if (processedChannels == numInputChannels) {
454  return true;
455  }
456  }
457  }
458 
459  else {
460  auto d2 = std::div(numInputChannels, 2);
461  for (int b = 0;
462  b < std::min(d2.quot + (d2.rem > 0 ? 1 : 0), (int)buffers2.size());
463  ++b) {
464  int r = std::min(2, numInputChannels - processedChannels);
465  for (int i = 0; i < r; ++i) {
466  auto c = b * 2 + i;
467  for (int j = 0; j < numInputSamples; ++j) {
468  buffers2[b](j * 2 + i) = input[c][j];
469  }
470  }
471  processedChannels += r;
472  assert(processedChannels <= numInputChannels);
473  if (processedChannels == numInputChannels) {
474  return true;
475  }
476  }
477  }
478 
479  assert(false);
480  return false;
481 }

485 InterleavedBuffer<Scalar>::At(int channel, int sample) const
486 {
487  if constexpr (VEC8_AVAILABLE) {
488 #if AVEC_MIX_VEC_SIZES

490  if (d8.quot < buffers8.size()) {
491  return &buffers8[d8.quot](8 * sample + d8.rem);
492  }
493  else {
494  return &buffers4[0](4 * sample + d8.rem);
495  }
496 #else
497  auto d8 = std::div(channel, 8);
498  return &buffers8[d8.quot](8 * sample + d8.rem);
499 #endif
500  }
501  else if constexpr (VEC4_AVAILABLE) {
502  auto d4 = std::div(channel, 4);
503  return &buffers4[d4.quot](4 * sample + d4.rem);
504  }
505  else {
506  auto d2 = std::div(channel, 2);
507  return &buffers2[d2.quot](2 * sample + d2.rem);

537  &buffers8[i](0));
538  numChannelsToCopy -= 8;
