251 template<typename Scalar>
252 inline void
253 getNumOfVecBuffersUsedByInterleavedBuffer(int numChannels,
254  int& num2,
255  int& num4,
256  int& num8)
257 {
258  constexpr bool VEC8_AVAILABLE = SimdTypes<Scalar>::VEC8_AVAILABLE;
259  constexpr bool VEC4_AVAILABLE = SimdTypes<Scalar>::VEC4_AVAILABLE;
260  constexpr bool VEC2_AVAILABLE = SimdTypes<Scalar>::VEC2_AVAILABLE;
261  if constexpr (VEC8_AVAILABLE) {
262  if (numChannels <= 4) {
263  num4 = 1;
264  num8 = num2 = 0;
265  }
266  else {
267  auto d8 = std::div(numChannels, 8);
268  num8 = (int)d8.quot + (d8.rem > 4 ? 1 : 0);
269  num4 = (d8.rem > 0 && d8.rem <= 4) ? 1 : 0;
270  num2 = 0;
271  }
272  }
273  else if constexpr (VEC4_AVAILABLE) {
274  auto d4 = std::div(numChannels, 4);
275  num8 = 0;
276  if constexpr (VEC2_AVAILABLE) {
277  if (numChannels <= 2) {
278  num2 = 1;
279  num4 = 0;
280  }
281  else {
282  num4 = (int)d4.quot + (d4.rem > 2 ? 1 : 0);
283  num2 = (d4.rem > 0 && d4.rem <= 2) ? 1 : 0;
284  }
285  }
286  else {
287  num4 = (int)d4.quot + (d4.rem > 0 ? 1 : 0);
288  num2 = 0;
289  }
290  }
291  else {
292  auto d2 = std::div(numChannels, 2);
293  num8 = 0;
294  num2 = (int)d2.quot + (d2.rem > 0 ? 1 : 0);
295  num4 = 0;
296  }

320  static auto doAtChannel(int channel, T2& v2, T4& v4, T8& v8, Action action)
321  {
322  constexpr bool VEC8_AVAILABLE = SimdTypes<Scalar>::VEC8_AVAILABLE;
323  constexpr bool VEC4_AVAILABLE = SimdTypes<Scalar>::VEC4_AVAILABLE;

325 
326  if constexpr (VEC8_AVAILABLE) {
327  if (v4.size() > 0) {
328  if (channel < 4) {
329  return action(v4[0], channel, 4);
330  }
331  else {
332  auto d8 = std::div(channel - 4, 8);
333  return action(v8[d8.quot], d8.rem, 8);

338  return action(v8[d8.quot], d8.rem, 8);
339  }

342  if constexpr (VEC2_AVAILABLE) {

344  if (channel < 2) {
345  return action(v2[0], channel, 2);
346  }
347  else {

349  return action(v4[d4.quot], d4.rem, 4);

442 {
443  if (numOutputChannels > numChannels || numOutputSamples > numSamples) {
444  return false;
445  }
446

448 
449  if constexpr (VEC2_AVAILABLE) {
450  if (buffers2.size() > 0) {
451  auto d2 = std::div(numOutputChannels, 2);
452  for (int b = 0;
453  b < std::min(d2.quot + (d2.rem > 0 ? 1 : 0), (int)buffers2.size());
454  ++b) {
455  int r = std::min(2, numOutputChannels - processedChannels);
456  for (int i = 0; i < r; ++i) {
457  auto c = i + processedChannels;
458  for (int j = 0; j < numOutputSamples; ++j) {
459  output[c][j] = buffers2[b](j * 2 + i);
460  }
461  }
462  processedChannels += r;
463  assert(processedChannels <= numOutputChannels);
464  if (processedChannels == numOutputChannels) {
465  return true;
466  }
467  }
468  }
469  }

474  b < std::min(d4.quot + (d4.rem > 0 ? 1 : 0), (int)buffers4.size());
