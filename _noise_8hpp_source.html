363 template<typename Scalar>
364 inline NoiseGenerator<Scalar>::NoiseGenerator(int numChannels, uint16_t seed)
365  : numChannels(numChannels)
366 {
367  int num8, num4, num2;
368  if constexpr (VEC8_AVAILABLE) {
369  auto d8 = std::div(numChannels, 8);
370 #if AVEC_MIX_VEC_SIZES
371  num8 = (int)d8.quot + (d8.rem > 4 ? 1 : 0);
372  num4 = (d8.rem <= 4) ? 1 : 0;
373 #else
374  num8 = (int)d8.quot + (d8.rem > 0 ? 1 : 0);
375  num4 = 0;
376 #endif
377  num2 = 0;
378  }
379  else if constexpr (VEC4_AVAILABLE) {
380  auto d4 = std::div(numChannels, 4);
381  num8 = 0;
382  num4 = (int)d4.quot + (d4.rem > 0 ? 1 : 0);
383  num2 = 0;
384  }
385  else {
386  auto d4 = std::div(numChannels, 4);
387  num8 = 0;
388  num2 = (int)d4.quot + (d4.rem > 0 ? 1 : 0);
389  num4 = 0;
390  }
391  generators8.reserve(num8);
392  generators4.reserve(num4);
393  generators2.reserve(num2);
394  uint16_t s = seed;
395  for (int i = 0; i < num8; ++i) {
396  generators8.push_back(
397  VecNoiseGenerator<Vec8>({ s++, s++, s++, s++, s++, s++, s++, s++ }));
398  }
399  for (int i = 0; i < num4; ++i) {
400  generators4.push_back(VecNoiseGenerator<Vec4>({ s++, s++, s++, s++ }));
401  }
402  for (int i = 0; i < num2; ++i) {
403  generators2.push_back(VecNoiseGenerator<Vec2>({ s++, s++, s++, s++ }));
404  }
405 }

484 template<typename Scalar>
485 inline void
486 NoiseGenerator<Scalar>::Generate(InterleavedBuffer<Scalar>& outputBuffer,
487  int numSamples,
488  int numChannelsToGenerate)
489 {
490  assert(numChannelsToGenerate <= numChannels);
491 
492  outputBuffer.SetNumSamples(numSamples);
493  if constexpr (VEC8_AVAILABLE) {
494  for (int i = 0; i < generators8.size(); ++i) {
495  generators8[i].Generate(outputBuffer.GetBuffer8(i), numSamples);
496  numChannelsToGenerate -= 8;
497  if (numChannelsToGenerate <= 0) {
498  return;
499  }
500  }
501  if (generators4.size() > 0) {
502  generators4[0].Generate(outputBuffer.GetBuffer4(0), numSamples);
503  numChannelsToGenerate -= 4;
504  }
505  }
506  else if constexpr (VEC4_AVAILABLE) {
507  for (int i = 0; i < generators4.size(); ++i) {
508  generators4[i].Generate(outputBuffer.GetBuffer4(i), numSamples);
509  numChannelsToGenerate -= 4;
510  if (numChannelsToGenerate <= 0) {
511  return;
512  }
513  }
514  }
515  else {
516  int lastBuffers2 = outputBuffer.GetNumBuffers2() - 1;
517  for (int i = 0; i < generators2.size(); ++i) {
518  VecBuffer<Vec2>* next =
519  (i < lastBuffers2) ? next : outputBuffer.GetBuffer2(i + 1);
520  generators2[i].Generate(outputBuffer.GetBuffer2(i), numSamples, next);
521  numChannelsToGenerate -= 4;
522  if (numChannelsToGenerate <= 0) {
523  return;
524  }
525  }
526  }
527 }
