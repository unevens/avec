102  void setNumChannels(int numRequiredChannels)
103  {
104  data.resize(numRequiredChannels);
105  for (int i = 0; i < data.size(); ++i) {

116  void reserve(int numSamples)
117  {
118  if (capacity >= numSamples) {
119  return;
120  }
121  capacity = numSamples;
122  for (int i = 0; i < data.size(); ++i) {
123  data[i].reserve(numSamples);
124  }
125  updatePointers();
126  }

134  void setNumSamples(int numSamples, bool shrinkIfSmaller = false)
135  {
136  reserve(numSamples);
137  size = numSamples;
138  for (int i = 0; i < data.size(); ++i) {
139  data[i].resize(numSamples, 0.0);
140  }
141  if (shrinkIfSmaller) {
142  shrink();
143  }
144  updatePointers();
145  }

167  void shrink()
168  {
169  data.shrink_to_fit();
170  for (int i = 0; i < data.size(); ++i) {
171  data[i].shrink_to_fit();
172  }
173  updatePointers();
174  pointers.shrink_to_fit();
175  capacity = size;
176  }

190 template<typename InScalar, typename OutScalar>
191 inline void
192 copyScalarBuffer(ScalarBuffer<InScalar> const& input,
193  ScalarBuffer<OutScalar>& output,
194  int numChannels = -1)
195 {
196  if (numChannels < 0) {
197  numChannels = input.getNumChannels();
198  }
199  output.setNumChannelsAndSamples(numChannels, input.getNumSamples());
200  for (int c = 0; c < numChannels; ++c) {
201  std::copy(
202  &input[c][0], &input[c][0] + input.getNumSamples(), &output[c][0]);
203  }
204 }
