116  void Reserve(int requiredSize)
117  {
118  if (capacity >= requiredSize) {
119  return;
120  }
121  capacity = requiredSize;

134  void SetSize(int requiredSize, bool shrink = false)
135  {

186  {

208 template<typename InScalar, typename OutScalar>
209 inline void
210 CopyScalarBuffer(ScalarBuffer<InScalar> const& input,
211  ScalarBuffer<OutScalar>& output,
212  int numChannels = -1)
213 {
214  if (numChannels < 0) {
215  numChannels = input.GetNumChannels();
216  }
217  output.SetNumChannelsAndSize(numChannels, input.GetSize());
218  for (int c = 0; c < numChannels; ++c) {
219  std::copy(&input[c][0], &input[c][0] + input.GetSize(), &output[c][0]);
220  }
221 }
