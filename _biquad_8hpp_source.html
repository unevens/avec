98  void ProcessBlock(VecBuffer<Vec> const& input,
99  VecBuffer<Vec>& output,
100  int numSamples)
101  {
102  Vec a1 = buffer[0];
103  Vec a2 = buffer[1];
104  Vec b0 = buffer[2];
105  Vec b1 = buffer[3];
106  Vec b2 = buffer[4];
107  Vec prev0 = buffer[5];
108  Vec prev1 = buffer[6];
109  for (int i = 0; i < numSamples; ++i) {
110  Vec in = input[i];
111  Vec next_buffer_0 = in - a1 * prev0 - a2 * prev1;
112  Vec out = b0 * next_buffer_0 + b1 * prev0 + b2 * prev1;
113  prev1 = prev0;
114  prev0 = next_buffer_0;
115  output[i] = out;
116  }
117  buffer[5] = prev0;
118  buffer[6] = prev1;
119  }

281  void Setup(int channel, bool reset = true) override
282  {
283  double a1 = 0.0;
284  double a2 = 0.0;
285  double b0 = 0.0;
286  double b1 = 0.0;
287  double b2 = 0.0;
288  double g = pow(10.0, gain[channel] / 40.0);
289  double cs = cos(frequency[channel]);
290  double sn = sin(frequency[channel]);
291  double alpha = sn / (2.0 * quality[channel]);
292  double sq = 2.0 * sqrt(g) * alpha;
293  switch (filterType[channel]) {
294  case BiquadFilterType::HighShelf: {
295  double a0inv = 1.0 / ((g + 1.0) - (g - 1.0) * cs + sq);
296  b0 = a0inv * g * ((g + 1.0) + (g - 1.0) * cs + sq);
297  b1 = -2.0 * a0inv * g * ((g - 1.0) + (g + 1.0) * cs);
298  b2 = a0inv * g * ((g + 1.0) + (g - 1.0) * cs - sq);
299  a1 = 2.0 * a0inv * ((g - 1.0) - (g + 1.0) * cs);
300  a2 = a0inv * ((g + 1.0) - (g - 1.0) * cs - sq);
301  } break;
302  case BiquadFilterType::HighPass: {
303  double a0inv = 1.0 / (1.0 + alpha);
304  b0 = a0inv * 0.5 * (1.0 + cs);
305  b1 = a0inv * (-1.0 - cs);
306  b2 = a0inv * 0.5 * (1.0 + cs);
307  a1 = -2.0 * a0inv * cs;
308  a2 = (1.0 - alpha) * a0inv;
309  } break;
310  case BiquadFilterType::LowShelf: {
311  double a0inv = 1.0 / ((g + 1.0) + (g - 1.0) * cs + sq);
312  b0 = a0inv * g * ((g + 1.0) - (g - 1.0) * cs + sq);
313  b1 = 2.0 * a0inv * g * ((g - 1.0) - (g + 1.0) * cs);
314  b2 = a0inv * g * ((g + 1.0) - (g - 1.0) * cs - sq);
315  a1 = -2.0 * a0inv * ((g - 1.0) + (g + 1.0) * cs);
316  a2 = a0inv * ((g + 1.0) + (g - 1.0) * cs - sq);
317  } break;
318  case BiquadFilterType::LowPass: {
319  double a0inv = 1.0 / (1.0 + alpha);
320  b0 = a0inv * (1.0 - cs) * 0.5;
321  b1 = a0inv * (1.0 - cs);
322  b2 = a0inv * (1.0 - cs) * 0.5;
323  a1 = a0inv * (-2.0 * cs);
324  a2 = a0inv * (1.0 - alpha);
325  } break;
326  case BiquadFilterType::Peak: {
327  double a0inv = 1.0 / (1.0 + alpha / g);
328  b0 = a0inv * (1.0 + alpha * g);
329  b1 = a0inv * (-2.0 * cs);
330  b2 = a0inv * (1.0 - alpha * g);
331  a1 = a0inv * (-2.0 * cs);
332  a2 = a0inv * (1.0 - alpha / g);
333  } break;
334  case BiquadFilterType::BandPass: {
335  double a0inv = 1.0 / (1.0 + alpha);
336  b0 = a0inv * alpha;
337  b1 = 0.0;
338  b2 = -a0inv * alpha;
339  a1 = -2.0 * a0inv * cs;
340  a2 = a0inv * (1.0 - alpha);
341  } break;
342  case BiquadFilterType::Notch: {
343  double a0inv = 1.0 / (1.0 + alpha);
344  b0 = a0inv;
345  b1 = -2.0 * a0inv * cs;
346  b2 = a0inv;
347  a1 = -2.0 * a0inv * cs;
348  a2 = a0inv * (1.0 - alpha);
349  } break;
350  case BiquadFilterType::AllPass: {
351  double a0inv = 1.0 / (1.0 + alpha);
352  b0 = a0inv * (1.0 - alpha);
353  b1 = -2.0 * a0inv * cs;
354  b2 = a0inv * (1.0 + alpha);
355  a1 = -2.0 * a0inv * cs;
356  a2 = a0inv * (1.0 - alpha);
357  } break;
358  default:
359  assert(false);
360  }
361  buffer[0][channel] = a1;
362  buffer[1][channel] = a2;
363  buffer[2][channel] = b0;
364  buffer[3][channel] = b1;
365  buffer[4][channel] = b2;
366 
367  if (reset) {
368  Reset(channel);
369  }
370  }

422 template<typename Scalar>
423 class BiquadFilter final
424 {
425  using Vec8 = typename SimdTypes<Scalar>::Vec8;
426  using Vec4 = typename SimdTypes<Scalar>::Vec4;
427  using Vec2 = typename SimdTypes<Scalar>::Vec2;
428  static constexpr bool VEC8_AVAILABLE = SimdTypes<Scalar>::VEC8_AVAILABLE;
429  static constexpr bool VEC4_AVAILABLE = SimdTypes<Scalar>::VEC4_AVAILABLE;
430  int numChannels;
431  std::vector<VecBiquadFilter<Vec8>> filters8;
432  std::vector<VecBiquadFilter<Vec4>> filters4;
433  std::vector<VecBiquadFilter<Vec2>> filters2;
434 
435 public:
446  BiquadFilter(int numChannels,
447  BiquadFilterType filterType_ = BiquadFilterType::LowPass,
448  double frequency_ = 0.1,
449  double quality_ = 0.79,
450  double gain_ = 0.0)
451  : numChannels(numChannels)
452  {
453  int num8, num4, num2;
454  if constexpr (VEC8_AVAILABLE) {
455  auto d8 = std::div(numChannels, 8);
456 #if AVEC_MIX_VEC_SIZES
457  num8 = (int)d8.quot + (d8.rem > 4 ? 1 : 0);
458  num4 = (d8.rem <= 4) ? 1 : 0;
459 #else
460  num8 = (int)d8.quot + (d8.rem > 0 ? 1 : 0);
461  num4 = 0;
462 #endif
463  num2 = 0;
464  }
465  else if constexpr (VEC4_AVAILABLE) {
466  auto d4 = std::div(numChannels, 4);
467  num8 = 0;
468  num4 = (int)d4.quot + (d4.rem > 0 ? 1 : 0);
469  num2 = 0;
470  }
471  else {
472  auto d4 = std::div(numChannels, 4);
473  num8 = 0;
474  num2 = (int)d4.quot + (d4.rem > 0 ? 1 : 0);
475  num4 = 0;
476  }
477  filters8.reserve(num8);
478  filters4.reserve(num4);
479  filters2.reserve(num2);
480  for (int i = 0; i < num8; ++i) {
481  filters8.push_back(
482  VecBiquadFilter<Vec8>(filterType_, frequency_, quality_, gain_));
483  }
484  for (int i = 0; i < num4; ++i) {
485  filters4.push_back(
486  VecBiquadFilter<Vec4>(filterType_, frequency_, quality_, gain_));
487  }
488  for (int i = 0; i < num2; ++i) {
489  filters2.push_back(
490  VecBiquadFilter<Vec2>(filterType_, frequency_, quality_, gain_));
491  }
492  }

501  void ProcessBlock(InterleavedBuffer<Scalar> const& input,
502  InterleavedBuffer<Scalar>& output,
503  int numSamples,
504  int numChannelsToProcess)
505  {
506  assert(numChannelsToProcess <= numChannels);
507  assert(numSamples <= input.GetNumSamples());
508 
509  output.SetNumSamples(numSamples);
510  if constexpr (VEC8_AVAILABLE) {
511  for (int i = 0; i < filters8.size(); ++i) {
512  filters8[i].ProcessBlock(
513  input.GetBuffer8(i), output.GetBuffer8(i), numSamples);
514  numChannelsToProcess -= 8;
515  if (numChannelsToProcess <= 0) {
516  return;
517  }
518  }
519  if (filters4.size() > 0) {
520  filters4[0].ProcessBlock(
521  input.GetBuffer4(0), output.GetBuffer4(0), numSamples);
522  numChannelsToProcess -= 4;
523  }
524  }
525  else if constexpr (VEC4_AVAILABLE) {
526  for (int i = 0; i < filters4.size(); ++i) {
527  filters4[i].ProcessBlock(
528  input.GetBuffer4(i), output.GetBuffer4(i), numSamples);
529  numChannelsToProcess -= 4;
530  if (numChannelsToProcess <= 0) {
531  return;
532  }
533  }
534  }
535  else {
536  int lastBuffers2 = output.GetNumBuffers2() - 1;
537  for (int i = 0; i < filters2.size(); ++i) {
538  filters2[i].ProcessBlock(
539  input.GetBuffer2(i), output.GetBuffer2(i), numSamples);
540  numChannelsToProcess -= 2;
541  if (numChannelsToProcess <= 0) {
542  return;
543  }
544  }
545  }
546  }
