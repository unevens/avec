70  void ProcessBlock(VecBuffer<Vec> const& input,
71  VecBuffer<Vec>& output,
72  int numSamples)
73  {
74  Vec a1 = buffer[0];
75  Vec a2 = buffer[1];
76  Vec b0 = buffer[2];
77  Vec b1 = buffer[3];
78  Vec b2 = buffer[4];
79  Vec prev0 = buffer[5];
80  Vec prev1 = buffer[6];
81  for (int i = 0; i < numSamples; ++i) {
82  Vec in = input[i];
83  Vec next_buffer_0 = in - a1 * prev0 - a2 * prev1;
84  Vec out = b0 * next_buffer_0 + b1 * prev0 + b2 * prev1;
85  prev1 = prev0;
86  prev0 = next_buffer_0;
87  output[i] = out;
88  }
89  buffer[5] = prev0;
90  buffer[6] = prev1;
91  }

218 private:
219  void Setup()
220  {
221  for (int i = 0; i < Vec::size(); ++i) {
222  double a1 = 0.0;
223  double a2 = 0.0;
224  double b0 = 0.0;
225  double b1 = 0.0;
226  double b2 = 0.0;
227  double g = pow(10.0, gain[i] / 40.0);
228  double cs = cos(frequency[i]);
229  double sn = sin(frequency[i]);
230  double alpha = sn / (2.0 * quality[i]);
231  double sq = 2.0 * sqrt(g) * alpha;
232  switch (filterType[i]) {
233  case BiquadFilterType::HighShelf: {
234  double a0inv = 1.0 / ((g + 1.0) - (g - 1.0) * cs + sq);
235  b0 = a0inv * g * ((g + 1.0) + (g - 1.0) * cs + sq);
236  b1 = -2.0 * a0inv * g * ((g - 1.0) + (g + 1.0) * cs);
237  b2 = a0inv * g * ((g + 1.0) + (g - 1.0) * cs - sq);
238  a1 = 2.0 * a0inv * ((g - 1.0) - (g + 1.0) * cs);
239  a2 = a0inv * ((g + 1.0) - (g - 1.0) * cs - sq);
240  } break;
241  case BiquadFilterType::HighPass: {
242  double a0inv = 1.0 / (1.0 + alpha);
243  b0 = a0inv * 0.5 * (1.0 + cs);
244  b1 = a0inv * (-1.0 - cs);
245  b2 = a0inv * 0.5 * (1.0 + cs);
246  a1 = -2.0 * a0inv * cs;
247  a2 = (1.0 - alpha) * a0inv;
248  } break;
249  case BiquadFilterType::LowShelf: {
250  double a0inv = 1.0 / ((g + 1.0) + (g - 1.0) * cs + sq);
251  b0 = a0inv * g * ((g + 1.0) - (g - 1.0) * cs + sq);
252  b1 = 2.0 * a0inv * g * ((g - 1.0) - (g + 1.0) * cs);
253  b2 = a0inv * g * ((g + 1.0) - (g - 1.0) * cs - sq);
254  a1 = -2.0 * a0inv * ((g - 1.0) + (g + 1.0) * cs);
255  a2 = a0inv * ((g + 1.0) + (g - 1.0) * cs - sq);
256  } break;
257  case BiquadFilterType::LowPass: {
258  double a0inv = 1.0 / (1.0 + alpha);
259  b0 = a0inv * (1.0 - cs) * 0.5;
260  b1 = a0inv * (1.0 - cs);
261  b2 = a0inv * (1.0 - cs) * 0.5;
262  a1 = a0inv * (-2.0 * cs);
263  a2 = a0inv * (1.0 - alpha);
264  } break;
265  case BiquadFilterType::Peak: {
266  double a0inv = 1.0 / (1.0 + alpha / g);
267  b0 = a0inv * (1.0 + alpha * g);
268  b1 = a0inv * (-2.0 * cs);
269  b2 = a0inv * (1.0 - alpha * g);
270  a1 = a0inv * (-2.0 * cs);
271  a2 = a0inv * (1.0 - alpha / g);
272  } break;
273  case BiquadFilterType::BandPass: {
274  double a0inv = 1.0 / (1.0 + alpha);
275  b0 = a0inv * alpha;
276  b1 = 0.0;
277  b2 = -a0inv * alpha;
278  a1 = -2.0 * a0inv * cs;
279  a2 = a0inv * (1.0 - alpha);
280  } break;
281  case BiquadFilterType::Notch: {
282  double a0inv = 1.0 / (1.0 + alpha);
283  b0 = a0inv;
284  b1 = -2.0 * a0inv * cs;
285  b2 = a0inv;
286  a1 = -2.0 * a0inv * cs;
287  a2 = a0inv * (1.0 - alpha);
288  } break;
289  case BiquadFilterType::AllPass: {
290  double a0inv = 1.0 / (1.0 + alpha);
291  b0 = a0inv * (1.0 - alpha);
292  b1 = -2.0 * a0inv * cs;
293  b2 = a0inv * (1.0 + alpha);
294  a1 = -2.0 * a0inv * cs;
295  a2 = a0inv * (1.0 - alpha);
296  } break;
297  default:
298  assert(false);
299  }
300  buffer[0][i] = a1;
301  buffer[1][i] = a2;
302  buffer[2][i] = b0;
303  buffer[3][i] = b1;
304  buffer[4][i] = b2;
305  }
306  Reset();
307  }

323  using Vec4 = typename SimdTypes<Scalar>::Vec4;
324  using Vec2 = typename SimdTypes<Scalar>::Vec2;

332 public:
343  BiquadFilter(int numChannels,
344  BiquadFilterType filterType_ = BiquadFilterType::LowPass,
345  double frequency_ = 0.1,
346  double quality_ = 0.79,
347  double gain_ = 0.0)
348  : numChannels(numChannels)
349  {
350  int num8, num4, num2;
351  if constexpr (VEC8_AVAILABLE) {
352  auto d8 = std::div(numChannels, 8);
353 #if AVEC_MIX_VEC_SIZES
354  num8 = (int)d8.quot + (d8.rem > 4 ? 1 : 0);
355  num4 = (d8.rem <= 4) ? 1 : 0;
356 #else
357  num8 = (int)d8.quot + (d8.rem > 0 ? 1 : 0);
358  num4 = 0;
359 #endif
360  num2 = 0;
361  }
362  else if constexpr (VEC4_AVAILABLE) {
363  auto d4 = std::div(numChannels, 4);
364  num8 = 0;
365  num4 = (int)d4.quot + (d4.rem > 0 ? 1 : 0);
366  num2 = 0;
367  }
368  else {
369  auto d4 = std::div(numChannels, 4);
370  num8 = 0;
371  num2 = (int)d4.quot + (d4.rem > 0 ? 1 : 0);
372  num4 = 0;
373  }
374  filters8.reserve(num8);
375  filters4.reserve(num4);
376  filters2.reserve(num2);
377  for (int i = 0; i < num8; ++i) {
378  filters8.push_back(
379  VecBiquadFilter<Vec8>(filterType_, frequency_, quality_, gain_));
380  }
381  for (int i = 0; i < num4; ++i) {
382  filters4.push_back(
383  VecBiquadFilter<Vec4>(filterType_, frequency_, quality_, gain_));
384  }
385  for (int i = 0; i < num2; ++i) {
386  filters2.push_back(
387  VecBiquadFilter<Vec2>(filterType_, frequency_, quality_, gain_));
388  }
389  }

398  void ProcessBlock(InterleavedBuffer<Scalar> const& input,
399  InterleavedBuffer<Scalar>& output,
400  int numSamples,
401  int numChannelsToProcess)
402  {
403  assert(numChannelsToProcess <= numChannels);
404  assert(numSamples <= input.GetNumSamples());
405 
406  output.SetNumSamples(numSamples);
407  if constexpr (VEC8_AVAILABLE) {
408  for (int i = 0; i < filters8.size(); ++i) {
409  filters8[i].ProcessBlock(
410  input.GetBuffer8(i), output.GetBuffer8(i), numSamples);
411  numChannelsToProcess -= 8;
412  if (numChannelsToProcess <= 0) {
413  return;
414  }
415  }
416  if (filters4.size() > 0) {
417  filters4[0].ProcessBlock(
418  input.GetBuffer4(0), output.GetBuffer4(0), numSamples);
419  numChannelsToProcess -= 4;
420  }
421  }
422  else if constexpr (VEC4_AVAILABLE) {
423  for (int i = 0; i < filters4.size(); ++i) {
424  filters4[i].ProcessBlock(
425  input.GetBuffer4(i), output.GetBuffer4(i), numSamples);
426  numChannelsToProcess -= 4;
427  if (numChannelsToProcess <= 0) {
428  return;
429  }
430  }
431  }
432  else {
433  int lastBuffers2 = output.GetNumBuffers2() - 1;
434  for (int i = 0; i < filters2.size(); ++i) {
435  filters2[i].ProcessBlock(
436  input.GetBuffer2(i), output.GetBuffer2(i), numSamples);
437  numChannelsToProcess -= 2;
438  if (numChannelsToProcess <= 0) {
439  return;
440  }
441  }
442  }
443  }
